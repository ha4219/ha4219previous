#include <iostream>
#include <queue>
#include <cmath>

using namespace std;

int boj_18224_map[500][500];
int boj_18224_v[500][500][11][2];

typedef struct boj_18224_s{
    int x;
    int y;
    int c;
}boj_18224_s;

void boj_18224(){
    int n, m;
    cin>>n>>m;
    int dx[] = {1, -1, 0, 0};
    int dy[] = {0, 0, 1, -1};
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            cin>>boj_18224_map[i][j];
        }
    }
    int res=-1;
    queue<boj_18224_s> q;
    q.push({0,0,1});
    boj_18224_v[0][0][1][1] = 1;
    while (!q.empty()) {
        int x = q.front().x, y = q.front().y, c = q.front().c;
        q.pop();
        if (x == n-1 && y == n-1){
            res = c;
            break;
        }
        for(int i=0;i<4;i++){
            int cx = x+dx[i], cy = y+dy[i];
            if (cx>=n || cx<0 || cy>=n || cy<0){
                continue;
            }
            int t=(c/m)%2;
            if (boj_18224_v[cx][cy][c%m][t]){
                continue;
            }
            if (t==0){
                if (!boj_18224_map[cx][cy]){
                    boj_18224_v[cx][cy][c%m][t] = 1;
                    q.push({cx, cy, c+1});
                }
            }else{
                if (!boj_18224_map[cx][cy]){
                    boj_18224_v[cx][cy][c%m][t] = 1;
                    q.push({cx, cy, c+1});
                }
                else{
                    while (true) {
                        cx+=dx[i];
                        cy+=dy[i];
                        if (cx<0||cy<0||cx>=n||cy>=n||boj_18224_v[cx][cy][c%m][t]){
                            break;
                        }else if(boj_18224_map[cx][cy]==0){
                            boj_18224_v[cx][cy][c%m][t] = 1;
                            q.push({cx, cy, c+1});
                            break;
                        }
                    }
                }
            }
        }
    }
    if (res == -1){
        cout<<res<<endl;
    }else{
        if ((res/m)%2){
            cout<<res/(m*2) + 1<<" moon"<<'\n';
        }else{
            cout<<res/(m*2) + 1<<" sun"<<'\n';
        }
    }
}

int main(){
    cin.sync_with_stdio(false);
    cin.tie(NULL);
    boj_18224();
    return 0;
}
