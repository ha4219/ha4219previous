from sys import stdin

n, m = map(int, stdin.readline().split())
a = [list(map(int, stdin.readline().split())) for _ in range(n)]


def f():
    fir = a[0]
    for i in range(1, n):
        for j in range(m):
            if not fir[j] and a[i][j]:
                return False
        fir = a[i]
    return True


print("YES" if f() else "NO")
