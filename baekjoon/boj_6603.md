~~~python3
import sys
from itertools import combinations


def show(arr):
    for t in arr:
        print(t, end=' ')
    print()

    
while(True):
    a = sys.stdin.readline()
    a = a.split()
    n = int(a[0])

    if n==0:
        break
        
    arr = [int(a[i+1]) for i in range(n)]


    tmp = list(combinations(arr, 6))
    for tm in tmp:
        show(tm)
    print()
~~~
