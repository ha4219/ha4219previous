#include <iostream>
typedef long long ll;
using namespace std;


void boj_1722(){
    int n,m;
    cin>>n>>m;
    ll f[21]={1,};
    int v[21]={0,};
    for (int i=1;i<=n;i++){
        f[i]=i*f[i-1];
    }
    if(m&1){
        ll k;
        cin>>k;
        int cnt;
        for(int i=1;i<=n;i++){
            cnt=1;
            for(int j=1;j<=n;j++){
                if(v[j])
                    continue;
                if(cnt*f[n-i]>=k){
                    v[j]=1;
                    cout<<j<<" ";
                    k-=(cnt-1)*f[n-i];
                    break;
                }
                cnt++;
            }
        }
        cout<<endl;
    }else{
        ll arr[21]={0,};
        for(int i=1;i<=n;i++){
            cin>>arr[i];
        }
        ll k=1;
        int cnt;
        for(int i=1;i<=n;i++){
            cnt=1;
            for(int j=1;j<=n;j++){
                if(v[j])
                    continue;
                if(arr[i]==j){
                    v[j]=1;
                    k+=(cnt-1)*f[n-i];
                    break;
                }
                cnt++;
            }
        }
        cout<<k<<endl;
    }
}

int main(){
    cin.sync_with_stdio(false);
    cin.tie(NULL);
    boj_1722();
    return 0;
}
