from sys import stdin
from collections import deque

n = int(stdin.readline().strip())
m = int(stdin.readline().strip())
a = []

def ch(n):
    for s, c in a:
        if s == n:
            return True
    return False


def f_i(n):
    i = 0
    for s, c in a:
        if s == n:
            return i
        i += 1

def inc(n):
    i = f_i(n)
    a[i][1] += 1

def ad(t):
    if len(a) == n:
        r()
    a.append([t, 1])

def r():
    i = f_m_c()
    a.pop(i[0])

def f_m_c():
    t = []
    t_c = float('inf')
    i = 0
    for s, c in a:
        if c < t_c:
            t = [i]
            t_c = c
        elif c == t_c:
            t.append(i)
        i += 1
    return t

for i in map(int, stdin.readline().split()):
    if ch(i):
        inc(i)
    else:
        ad(i)
print(*sorted(list(zip(*a))[0]))
